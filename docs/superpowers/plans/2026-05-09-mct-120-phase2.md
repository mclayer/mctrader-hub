```python
from __future__ import annotations

from dataclasses import dataclass
from decimal import Decimal
from typing import Literal, Protocol


@dataclass(frozen=True)
class TradeCost:
    fee: Decimal
    slippage_bps: int
    funding_fee: Decimal = Decimal("0")

    @property
    def total(self) -> Decimal:
        return self.fee + self.funding_fee


class CostModel(Protocol):
    def estimate(
        self,
        symbol: str,
        side: Literal["buy", "sell"],
        quantity: Decimal,
        price: Decimal,
        *,
        is_maker: bool = False,
    ) -> TradeCost: ...

    def net_edge(self, *, expected_return: Decimal, cost: TradeCost) -> Decimal: ...


class BacktestFeeModel:
    name = "backtest_fee"

    def __init__(
        self,
        *,
        taker_rate: Decimal = Decimal("0.0004"),
        maker_rate: Decimal = Decimal("0.0002"),
        slippage_bps: int = 5,
    ) -> None:
        self._taker = taker_rate
        self._maker = maker_rate
        self._slippage_bps = slippage_bps

    def estimate(
        self,
        symbol: str,
        side: Literal["buy", "sell"],
        quantity: Decimal,
        price: Decimal,
        *,
        is_maker: bool = False,
    ) -> TradeCost:
        notional = quantity * price
        rate = self._maker if is_maker else self._taker
        return TradeCost(fee=notional * rate, slippage_bps=0 if is_maker else self._slippage_bps)

    def net_edge(self, *, expected_return: Decimal, cost: TradeCost) -> Decimal:
        return expected_return - cost.total


class LiveFeeModel:
    name = "live_fee"

    def __init__(
        self,
        *,
        taker_rate: Decimal = Decimal("0.0004"),
        maker_rate: Decimal = Decimal("0.0002"),
    ) -> None:
        self._taker = taker_rate
        self._maker = maker_rate

    def estimate(
        self,
        symbol: str,
        side: Literal["buy", "sell"],
        quantity: Decimal,
        price: Decimal,
        *,
        is_maker: bool = False,
    ) -> TradeCost:
        notional = quantity * price
        rate = self._maker if is_maker else self._taker
        return TradeCost(fee=notional * rate, slippage_bps=10)

    def net_edge(self, *, expected_return: Decimal, cost: TradeCost) -> Decimal:
        return expected_return - cost.total
```

```python
from __future__ import annotations

from dataclasses import dataclass
from decimal import Decimal, ROUND_DOWN
from typing import Protocol

from mctrader_engine.pipeline.types import OrderIntent

_8DP = Decimal("0.00000001")
_MIN_NOTIONAL_BITHUMB = Decimal("5000")  # KRW 기준, USD 기준 약 $5


@dataclass(frozen=True)
class NormalizedIntent:
    symbol: str
    side: str
    quantity: Decimal
    limit_price: Decimal | None
    rejected: bool
    reject_reason: str | None


class ExchangeSymbolConstraints(Protocol):
    min_notional_usd: Decimal
    lot_size_decimals: int
    min_qty: Decimal

    def normalize_qty(self, qty: Decimal) -> Decimal: ...


class BithumbConstraints:
    min_notional_usd: Decimal = Decimal("5")
    lot_size_decimals: int = 8
    min_qty: Decimal = Decimal("0.00000001")

    def normalize_qty(self, qty: Decimal) -> Decimal:
        precision = Decimal("1e-" + str(self.lot_size_decimals))
        return qty.quantize(precision, rounding=ROUND_DOWN)


class ExchangeConstraintValidator:
    def __init__(self, *, constraints: ExchangeSymbolConstraints) -> None:
        self._constraints = constraints

    def validate_and_normalize(
        self, intent: OrderIntent, *, market_price: Decimal
    ) -> NormalizedIntent:
        normalized_qty = self._constraints.normalize_qty(intent.quantity)

        if normalized_qty < self._constraints.min_qty:
            return NormalizedIntent(
                symbol=intent.symbol, side=intent.side,
                quantity=normalized_qty, limit_price=intent.limit_price,
                rejected=True, reject_reason="below_min_qty",
            )

        notional = normalized_qty * market_price
        if notional < self._constraints.min_notional_usd:
            return NormalizedIntent(
                symbol=intent.symbol, side=intent.side,
                quantity=normalized_qty, limit_price=intent.limit_price,
                rejected=True,
                reject_reason=f"min_notional:{notional:.4f}<{self._constraints.min_notional_usd}",
            )

        return NormalizedIntent(
            symbol=intent.symbol, side=intent.side,
            quantity=normalized_qty, limit_price=intent.limit_price,
            rejected=False, reject_reason=None,
        )
```

```python
from __future__ import annotations

from dataclasses import dataclass
from decimal import Decimal
from enum import Enum
from typing import Literal
from uuid import UUID, uuid4

from mctrader_engine.pipeline.types import Signal


class SignalGroupAtomicity(str, Enum):
    ALL_OR_NONE = "all_or_none"
    BEST_EFFORT = "best_effort"
    HEDGE_REQUIRED = "hedge_required"


@dataclass(frozen=True)
class SignalGroup:
    group_id: UUID
    atomicity: SignalGroupAtomicity
    legs: tuple[Signal, ...]
    max_leg_skew_ms: int
    confidence: Decimal

    def is_valid(self) -> bool:
        return self.confidence >= Decimal("0.6") and len(self.legs) >= 2


@dataclass(frozen=True)
class PairSignalGroup(SignalGroup):
    long_signal: Signal
    short_signal: Signal
    spread_zscore: Decimal

    def __new__(
        cls,
        *,
        long_signal: Signal,
        short_signal: Signal,
        spread_zscore: Decimal,
        confidence: Decimal,
        max_leg_skew_ms: int = 500,
    ) -> "PairSignalGroup":
        instance = object.__new__(cls)
        object.__setattr__(instance, "group_id", uuid4())
        object.__setattr__(instance, "atomicity", SignalGroupAtomicity.ALL_OR_NONE)
        object.__setattr__(instance, "legs", (long_signal, short_signal))
        object.__setattr__(instance, "max_leg_skew_ms", max_leg_skew_ms)
        object.__setattr__(instance, "confidence", confidence)
        object.__setattr__(instance, "long_signal", long_signal)
        object.__setattr__(instance, "short_signal", short_signal)
        object.__setattr__(instance, "spread_zscore", spread_zscore)
        return instance
```

```python
from __future__ import annotations

from dataclasses import dataclass, field
from datetime import datetime
from decimal import Decimal
from statistics import mean
from typing import Deque
from collections import deque

from mctrader_engine.pipeline.types import PipelineFrame


@dataclass
class PipelineMetrics:
    total_frames: int
    last_frame_latency_ms: float
    avg_latency_ms: float
    p95_latency_ms: float
    risk_rejection_count: int
    signal_count_history: list[int]


class PipelineMonitor:
    def __init__(
        self,
        *,
        window_size: int = 100,
        signal_anomaly_threshold: float = 5.0,
    ) -> None:
        self._window_size = window_size
        self._anomaly_threshold = signal_anomaly_threshold
        self._latencies: Deque[float] = deque(maxlen=window_size)
        self._signal_counts: Deque[int] = deque(maxlen=window_size)
        self._risk_rejections: int = 0
        self._total_frames: int = 0

    def record_frame_completed(
        self,
        frame: PipelineFrame,
        *,
        start_ts: datetime,
        end_ts: datetime,
    ) -> None:
        latency_ms = (end_ts - start_ts).total_seconds() * 1000
        self._latencies.append(latency_ms)
        self._signal_counts.append(len(frame.signals))
        self._total_frames += 1

        if frame.risk_decision and not frame.risk_decision.allowed:
            self._risk_rejections += 1

    def has_signal_anomaly(
        self, *, current_signal_count: int, window_avg: float
    ) -> bool:
        if window_avg == 0:
            return False
        return current_signal_count > window_avg * self._anomaly_threshold

    def get_metrics(self) -> PipelineMetrics:
        latencies = list(self._latencies)
        avg = mean(latencies) if latencies else 0.0
        sorted_lat = sorted(latencies)
        p95_idx = int(len(sorted_lat) * 0.95)
        p95 = sorted_lat[p95_idx] if sorted_lat else 0.0

        return PipelineMetrics(
            total_frames=self._total_frames,
            last_frame_latency_ms=latencies[-1] if latencies else 0.0,
            avg_latency_ms=avg,
            p95_latency_ms=p95,
            risk_rejection_count=self._risk_rejections,
            signal_count_history=list(self._signal_counts),
        )
```

```python
from __future__ import annotations

from enum import Enum
from uuid import UUID


class HotSwapStage(str, Enum):
    ACTIVE = "active"
    PREPARED = "prepared"
    SHADOW = "shadow"
    RETIRED = "retired"


class HotSwapManager:
    def __init__(self, *, current_version_id: UUID) -> None:
        self._current = current_version_id
        self._candidate: UUID | None = None
        self._previous: UUID | None = None
        self._stage = HotSwapStage.ACTIVE

    @property
    def stage(self) -> HotSwapStage:
        return self._stage

    @property
    def current_version_id(self) -> UUID:
        return self._current

    @property
    def candidate_version_id(self) -> UUID | None:
        return self._candidate

    def prepare(self, *, candidate_version_id: UUID) -> None:
        self._candidate = candidate_version_id
        self._stage = HotSwapStage.PREPARED

    def activate(self) -> None:
        if self._candidate is None:
            raise RuntimeError("No candidate prepared — call prepare() first")
        self._previous = self._current
        self._current = self._candidate
        self._candidate = None
        self._stage = HotSwapStage.ACTIVE

    def rollback(self) -> None:
        if self._previous is not None:
            self._current = self._previous
            self._previous = None
        self._candidate = None
        self._stage = HotSwapStage.ACTIVE

    def retire(self) -> None:
        self._stage = HotSwapStage.RETIRED
```
