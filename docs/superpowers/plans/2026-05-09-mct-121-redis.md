# MCT-121: Redis Hot Cache 도입 Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** ingester가 WAL 기록 직후 Redis Stream/String에 최신 tick·orderbook을 publish하여 engine·web-api가 Parquet 파일 생성을 기다리지 않고 hot 데이터를 즉시 읽을 수 있게 한다.

**Architecture:** `RedisTickPublisher`를 CollectorDaemon에 선택적으로 주입. 트랜잭션은 Redis Stream (`XADD`, maxlen=1000)으로, orderbook snapshot은 Redis String (`SET`, JSON)으로 publish. 실패는 경고 로그만, WAL 기록은 절대 블로킹 안 함. hub compose.yml에 redis:7-alpine 추가.

**Tech Stack:** redis:7-alpine, redis[hiredis]>=5 (Python), mctrader-data collector 확장

```python
# src/mctrader_data/redis_publisher.py
"""Redis publisher — streams hot market events for real-time consumption."""
from __future__ import annotations

import json
import logging
import os
from typing import Any

log = logging.getLogger(__name__)

STREAM_MAXLEN = 1000


class RedisTickPublisher:
    """Publish market events to Redis Streams (transactions) and Strings (orderbook).

    Key schema:
      transaction stream : mctrader:stream:transaction:{exchange}:{symbol}
      orderbook snapshot : mctrader:ob:{exchange}:{symbol}

    Client is created lazily. All failures are caught and logged; WAL writing
    must never block on Redis availability.
    """

    def __init__(self, redis_url: str | None = None) -> None:
        self._url = redis_url or os.environ.get("REDIS_URL", "redis://redis:6379/0")
        self._client: Any = None

    def _get_client(self) -> Any:
        if self._client is None:
            import redis
            self._client = redis.from_url(self._url, decode_responses=True)
        return self._client

    def publish_transaction(self, exchange: str, symbol: str, record: dict) -> None:
        key = f"mctrader:stream:transaction:{exchange}:{symbol}"
        try:
            fields = {
                "price": str(record.get("price", "")),
                "quantity": str(record.get("quantity", "")),
                "side": str(record.get("side", "")),
                "ts_utc": str(record.get("ts_utc", "")),
            }
            self._get_client().xadd(key, fields, maxlen=STREAM_MAXLEN, approximate=True)
        except Exception:
            log.warning("[redis] publish_transaction failed key=%s", key, exc_info=True)

    def publish_orderbook_snapshot(self, exchange: str, symbol: str, record: dict) -> None:
        key = f"mctrader:ob:{exchange}:{symbol}"
        try:
            payload = json.dumps({
                "bids": record.get("bids", []),
                "asks": record.get("asks", []),
                "ts_utc": str(record.get("ts_utc", "")),
            })
            self._get_client().set(key, payload)
        except Exception:
            log.warning("[redis] publish_orderbook_snapshot failed key=%s", key, exc_info=True)
```

```python
from pathlib import Path
from unittest.mock import MagicMock

from mctrader_market.types import Symbol

from mctrader_data.collector import CollectorDaemon
from mctrader_data.redis_publisher import RedisTickPublisher


def _make_transaction_event():
    from mctrader_market_bithumb.ws_events import TransactionEvent
    evt = MagicMock(spec=TransactionEvent)
    evt.exchange = "bithumb"
    evt.symbol = Symbol.from_string("KRW-BTC")
    evt.price = 135000000
    evt.quantity = 0.001
    evt.side = "bid"

def _make_orderbook_snapshot_event():
    from mctrader_market_bithumb.ws_events import OrderbookSnapshotEvent

    lvl.price = 135000000
    lvl.quantity = 1.0
    evt.bids = [lvl]
    evt.asks = [lvl]
    evt.event_time = "2026-05-09T00:00:00+00:00"
    evt.received_at = "2026-05-09T00:00:00+00:00"
    evt.raw = None
    return evt


def test_collector_calls_publish_transaction(tmp_path):
    mock_publisher = MagicMock(spec=RedisTickPublisher)

        root=tmp_path,
        exchange="bithumb",
        symbol=Symbol.from_string("KRW-BTC"),

    assert call_kwargs["exchange"] == "bithumb"
```
