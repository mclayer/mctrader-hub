    Env: NAS_RETRY_QUEUE_PATH (default = /data/retry_queue/ — EC-2 박제 path,
    /data/wal hot path 침범 0).
    """

        self,
        path: Path,                      # NAS_RETRY_QUEUE_PATH

        metrics: PrometheusExporter | None = None,
    ) -> None: ...
